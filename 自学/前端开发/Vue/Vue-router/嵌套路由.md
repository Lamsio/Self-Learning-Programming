#### 嵌套路由
顾名思义，在原本的一级路由基础上，为特定的路由设置`children`从而实现子路由。

```js
export default new VueRouter({  
    routes: [  
        {  
            name: 'home',  
			path: '/',  
 component: Home  
 },  
 {  
            path: '/about',  
 component: About  
 },  
 {  
            path: '/home',  
 component: Home,  
 children: [  
                {  
                    path: 'news',  
 component: News  
 },  
 {  
                    path: 'message',  
 component: Message,  
 children: [  
                        {  
                            name: 'detail',  
 path: 'detail',  
 component: Detail  
 }  
                    ]  
                }  
            ]  
        }  
    ],  
 mode: "history"  
})
```
